import {
  View,
  Text,
  Alert,   //->

import { useWishlist } from "../contexts/wishlistContext";
import { useBag } from "../contexts/bagContext";

import Toast from "react-native-toast-message";

export default function ProductDetail() {
  const router = useRouter();
  const params = useLocalSearchParams();

  const { bag, addToBag } = useBag();
  const {user}=useUser();   //->
  const { wishlist, addToWishlist, removeFromWishlist } = useWishlist();

  const { id, name, price, img } = params ?? {};

  const [selectedSize, setSelectedSize] = useState(null);
  const [sizeError, setSizeError] = useState(false);

  const calculatedPrice =
    selectedSize === "Half KG" ? Number(price) / 2 : Number(price);

  const isInWishlist = wishlist.some((item) => String(item.id) === String(id));

  const handleToggleWishlist = () => {
    if (isInWishlist) {
      removeFromWishlist(id);
      Toast.show({ type: "info", text1: `${name} removed from wishlist` });
    } else {
      addToWishlist({
        id,
        name,
        price,
        img: { uri: img }, // backend image ✔
      });
      Toast.show({ type: "success", text1: `${name} added to wishlist` });
    }
  };

  if (!id) {
    return (
      <SafeAreaView style={styles.safeArea}>
        <View style={styles.center}>
          <Text>Product details not available.</Text>
        </View>
      </SafeAreaView>
    );
  }

// if (!user) {
//   Alert.alert("Login Required", "Please login to add items to your wishlist.");
//   return;
// }

  return (
    <SafeAreaView style={styles.safeArea}>
      <ScrollView contentContainerStyle={styles.scrollContent}>
        {/* HEADER */}
        <View style={styles.topRow}>
          <TouchableOpacity
            style={styles.backBtn}
            onPress={() => router.back()}
          >
            <Ionicons name="arrow-back" size={22} color="#22543d" />
          </TouchableOpacity>

          <View style={styles.topRightIcons}>
            <TouchableOpacity
              style={styles.iconBtn}
              onPress={() => router.push("/wishlist")}
            >
              <Ionicons name="heart-outline" size={26} color="#22543d" />
            </TouchableOpacity>

            <TouchableOpacity
              style={styles.iconBtn}
              onPress={() => router.push("/bag")}
            >
              <Ionicons name="bag-outline" size={26} color="#22543d" />
              {bag.length > 0 && (
                <View style={styles.bagBadge}>
                  <Text style={styles.bagBadgeText}>{bag.length}</Text>
                </View>
              )}
            </TouchableOpacity>
          </View>
        </View>

        {/* PRODUCT IMAGE - backend image ✔ */}
        <Image
          source={{ uri: img }}
          style={styles.productImage}
          resizeMode="contain"
        />

        {/* INFO BOX */}
        <View style={styles.infoBox}>
          <View style={styles.infoRow}>
            <View style={{ flex: 1 }}>
              <Text style={styles.productName}>{name}</Text>
              <Text style={styles.price}>₹{price}/kg</Text>
            </View>

            <TouchableOpacity
              onPress={handleToggleWishlist}
              style={styles.infoWishlistBtn}
            >
              <Ionicons
                name={isInWishlist ? "heart" : "heart-outline"}
                size={24}
                color={isInWishlist ? "red" : "#22543d"}
              />
            </TouchableOpacity>
          </View>

          {/* SIZE SELECTION */}
          <Text style={styles.sizeLabel}>Choose size:</Text>

          <View
            style={[
              styles.sizeRow,
              sizeError && {
                borderWidth: 1,
                borderColor: "red",
                padding: 6,
                borderRadius: 10,
              },
            ]}
          >
            {["Half KG", "Full KG"].map((size) => (
              <TouchableOpacity
                key={size}
                style={[
                  styles.sizeBtn,
                  selectedSize === size && styles.sizeBtnActive,
                ]}
                onPress={() => {
                  setSelectedSize(size);
                  setSizeError(false);
                }}
              >
                <Text
                  style={[
                    styles.sizeBtnText,
                    selectedSize === size && styles.sizeBtnTextActive,
                  ]}
                >
                  {size}
                </Text>
              </TouchableOpacity>
            ))}
          </View>

          {/* ADD TO BAG */}
          <TouchableOpacity
            style={styles.addToBagBtn}
            onPress={() => {
              if (!selectedSize) {
                setSizeError(true);
                Toast.show({
                  type: "error",
                  text1: "Please select a size",
                });
                return;
              }

             addToBag({
  id,
  // name,
  // price: calculatedPrice,
  // img: { uri: img },
  // qty: 1,
  // size: selectedSize,
});


              Toast.show({
                type: "success",
                text1: `${name} added to bag`,
              });

              setTimeout(() => {
                router.push("/bag");
              }, 600);
            }}
          >
            <Text style={styles.addToBagText}>Add to Bag</Text>
          </TouchableOpacity>
        </View>
      </ScrollView>
    </SafeAreaView>
  );
}

import React, { useState } from "react";
import {
  View,
  Text,
  StyleSheet,
  Image,
  ScrollView,
  TouchableOpacity,
  Alert,
} from "react-native";
import { SafeAreaView } from "react-native-safe-area-context";
import { Ionicons } from "@expo/vector-icons";
import { useRouter } from "expo-router";
import { useBag } from "../contexts/bagContext";
import { useUser } from "../contexts/userContext";

export default function Checkout() {
  const router = useRouter();
  const { bag } = useBag();
  const { user } = useUser();

  const FIXED_USER_ID = "693318d1e100ef27146bebc4"; // ⭐ Use this until login is ready

  const [paymentMode, setPaymentMode] = useState(null);

  const subtotal = bag.reduce((sum, item) => sum + item.price * item.qty, 0);
  const shipping = subtotal > 300 ? 0 : 40;
  const discount = subtotal > 500 ? 50 : 0;
  const total = subtotal + shipping - discount;

  // ============================
  //        HANDLE PAYMENT
  // ============================
  const handlePayment = async () => {
    if (!paymentMode) {
      Alert.alert("Select Payment Method", "Please choose a payment method.");
      return;
    }

    try {
      const orderData = {
        user: user?._id || user?.id || FIXED_USER_ID, // ⭐ always present
        products: bag.map((item) => ({
          product: item.id,
          quantity: item.qty,
        })),
        totalAmount: Number(total),
        address: "Village Muktsar, Punjab",
        paymentMethod: paymentMode,
        isPaid: paymentMode !== "COD",
      };

      console.log("ORDER SENDING:", orderData);

      const res = await fetch("http://172.21.115.85:5000/api/orders", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(orderData),
      });

      const data = await res.json();
      console.log("ORDER RESPONSE:", data);

      if (!res.ok) {
        Alert.alert("Order Failed", data.message || "Could not create order.");
        return;
      }

      // Redirect to payment screen with orderId
      router.push({
        pathname: "/payment",
        params: {
          paymentMode,
          total,
          orderId: data._id,
        },
      });
    } catch (error) {
      console.log("Order Create Error:", error);
      Alert.alert("Error", "Something went wrong while creating order.");
    }
  };

//   if (!user) {
//   Alert.alert("Login Required", "Please login to continue.");
//   return;
// }

  return (
    <SafeAreaView style={styles.safeArea}>
      {/* HEADER */}
      <View style={styles.header}>
        <TouchableOpacity onPress={() => router.back()} style={styles.backBtn}>
          <Ionicons name="arrow-back" size={24} color="#22543d" />
        </TouchableOpacity>
        <Text style={styles.title}>Checkout</Text>
      </View>

      <ScrollView showsVerticalScrollIndicator={false}>
        {/* ADDRESS */}
        <View style={styles.card}>
          <Text style={styles.sectionTitle}>Delivery Address</Text>
          <Text style={styles.addressText}>
            Riya Girdhar{"\n"}
            Sri Muktsar Sahib, Punjab{"\n"}
            India - 152026
          </Text>
        </View>

        {/* ORDER SUMMARY */}
        <View style={styles.card}>
          <Text style={styles.sectionTitle}>Order Summary</Text>
          {bag.map((item) => (
            <View key={item.id} style={styles.itemRow}>
              <Image source={item.img} style={styles.itemImg} />
              <View style={{ flex: 1 }}>
                <Text>{item.name}</Text>
                <Text>Qty: {item.qty}</Text>
              </View>
              <Text>₹{item.price * item.qty}</Text>
            </View>
          ))}
        </View>

        {/* BILL */}
        <View style={styles.card}>
          <Text>Subtotal: ₹{subtotal}</Text>
          <Text>Shipping: ₹{shipping}</Text>
          <Text>Discount: -₹{discount}</Text>
          <Text style={{ fontWeight: "bold", marginTop: 5 }}>
            Total: ₹{total}
          </Text>
        </View>

        {/* PAYMENT OPTIONS */}
        <View style={styles.card}>
          <Text style={styles.sectionTitle}>Select Payment Method</Text>

          {["COD", "UPI", "CARD"].map((mode) => (
            <TouchableOpacity
              key={mode}
              style={[
                styles.paymentOption,
                paymentMode === mode && styles.activePayment,
              ]}
              onPress={() => setPaymentMode(mode)}
            >
              <Text
                style={{
                  color: paymentMode === mode ? "#fff" : "#000",
                  fontWeight: "700",
                }}
              >
                {mode}
              </Text>
            </TouchableOpacity>
          ))}
        </View>
      </ScrollView>

      {/* PAY BUTTON */}
      <TouchableOpacity style={styles.payBtn} onPress={handlePayment}>
        <Text style={styles.payTxt}>Proceed to Payment</Text>
      </TouchableOpacity>
    </SafeAreaView>
  );
}

    export const addToCart = async (req, res) => {
  try {
    const { user, product, quantity = 1, size, price } = req.body;

    let cart = await Cart.findOne({ user });

    if (!cart) {
      cart = new Cart({
        user,
        products: [{ product, size, price, quantity }]
      });
    } else {
      const item = cart.products.find(
        (p) =>
          p.product.toString() === product &&
          p.size === size
      );

      if (item) {
        item.quantity += quantity;
      } else {
        cart.products.push({ product, size, price, quantity });
      }
    }

    await cart.save();

    const updated = await Cart.findOne({ user })
      .populate("products.product");

    res.json(updated);
  } catch (err) {
    res.status(500).json({ message: err.message });
  }
};
